# SettingsListItem

##### List item for the settings page. does not contain background.

## Props
- **`title`** - <RequiredText /> - `string` - The Title for the setting.
- **`subTitle`** - <RequiredText /> - `string` - The Sub-Title of the setting, should be the value.
- **`extra`** - <RequiredText /> - `ReactNode` - The buttons, dropdowns, inputs on the right side of the setting item.

## Import
```jsx
import { SettingsListItem } from "@terra-money/station-ui";
```

## Gas Adjustment
<div className="story_with_source_below">
  <Canvas of={SettingsListItemStories.GasAdjustment} />
  <Source code={`
    const [selectedValue, setSelectedValue] = useState("Custom")
const [customValue, setCustomValue] = useState(1.0)

return (
  <SettingsLineItem
    title={"Developer Mode"}
    subTitle={selectedValue}
    extra={
      <Flex gap={16} style={{ width: "unset" }}>
        <SettingsDropdown
          value={selectedValue}
          options={[
            { value: "Standard", label: "Standard 1.0" },
            { value: "Low", label: "Low 0.5" },
            { value: "High", label: "High 2.0" },
            { value: "Yolo", label: "Yolo 5.0" },
            { value: "Custom", label: "Custom" },
          ]}
          onChange={(value) => setSelectedValue(value)}
        />

        {selectedValue === "Custom" && (
          <Input
            placeholder={customValue.toString()}
            onChange={(e) => setCustomValue(Number(e.target.value))}
            wrapperClassName={styles.input__wrapper__override}
            className={styles.input__override}
          />
        )}

        <Button
          variant="primary"
          label={"Set"}
          onClick={() => {}}
          className={styles.button__override}
        />
      </Flex>
    }
  />
)
  `} />
</div>
